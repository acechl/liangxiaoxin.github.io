原生JS封装简单轮播图

轮播图是网页上经常用到的组件，在此利用原生JS封装一个简单的左右移动的轮播图组件（效果类似于淘宝首页轮播图）。

1.封装一个左右移动的运动函数（当用户触发某件事件时调用）
function animate(obj,target) {
    clearInterval(obj.timeId);//每次点击了按钮之后，先清空定时器
    obj.timeId = setInterval(function () {
        var step = 20;
        // 先获得demo对象的当前位置
        var leader = obj.offsetLeft;//先获得当前的距离左侧的位置
        step = leader < target ? step : -step;
        if (Math.abs(leader - target) > Math.abs(step)) {   // 距离大于step时触发
            leader = leader + step; // 赋值
            obj.style.left = leader + 'px'; //不断移动
        } else {
            clearInterval(obj.timeId);//清空定时器
            obj.style.left = target + 'px';//手动设置目标位置
        }

    }, 15);

for(var i=0;i<lis.length;i++){    //图片个数的长度

3.鼠标移到小方块上
var pic = 0; //图片的索引
var square = 0; //小方块的索引
//先让第一个小方块颜色
var olLis = ol.children;
olLis[0].className='current';
//onmouseover事件
for(var i=0;i<olLis.length;i++){
    olLis[i].index = i;
    olLis[i].onmouseover = function(){
        for(var j=0;j<olLis.length;j++){
            olLis[j].className= ""; 
        }
        this.className = "current";
        // 鼠标移入对应的小方块的时候，图片移动对应小方块索引个图片的宽度
        var target = -imgWidth*this.index;
        animate(ul,target);
        pic=square=this.index;
    }
}
4.鼠标移入显示左右按钮
//鼠标移入的时候，显示左右按钮
box.onmouseover = function(){
    arr.style.display = "block";
    clearInterval(timerId);

box.onmouseout = function(){
    arr.style.display="none";
    timerId=setInterval(playNext,1000);

5.左右按钮注册单击事件
//封装成playNext,可复用
function playNext(){
    if(pic==lis.length-1){
        pic=0;
        ul.style.left = "0px";
    }
    pic++;
    var target = -imgWidth*pic;
    animate(ul,target);
    if(square<olLis.length-1){
        square++;
    }else{
        square=0;
    }
    for(var j=0;j<olLis.length;j++){
        olLis[j].className='';
    }
    olLis[square].className = 'current';
}
//左右点击事件
arrRight.onclick = playNext;    
arrLeft.onclick = function(){
    if(pic==0){
        pic=lis.length-1;
        ul.style.left=-imgWidth*(lis.length-1)+'px';
    }
    pic--;
    var target = -imgWidth*pic;
    animate(ul,target);

    if(square>0){
        square--;
    }else {
        square = olLis.length-1;
    }
    for(var j=0;j<olLis.length;j++){
        olLis[j].className='';
    }
    olLis[square].className = 'current';
}
6.开启定时器，自动轮播
timerId = setInterval(playNext,1000);
