记录遇到的兼容性问题

透明度 opacity

//高级浏览器下直接用 opacity = xx;
opacity = 0.3; //透明度0.3 取值范围0~1；

//兼容IE8及之前版本；
filter = alpha(opacity=30); //透明度30% 取值范围0~100%；

获得标签之间的文本信息

function getText(element){  
        if(typeof element.innerText =="string"){ // 0,NaN,"",Null,undefined,false   
            return element.innerText; // IE8及之前的版本的浏览器支持的方式
        }else {
            return  element.textContent;//火狐的早期版本的浏览器支持的
        }
    }

设置标签之间的文本信息

function setText(element,value){
    // 有时候获得的内容有可能是“”
    if(typeof element.innerText =="string"){ // IE8及之前的版本的浏览器支持的方式
           element.innerText = value;
    }else {
        element.textContent = value;
    }
}

获得标签节点（封装在一个对象中）

//获得上一个兄弟标签节点
var elementNode = {
getPreviousElement:function(element){
    if(element.previousElementSibling){
        return element.previousElementSibling;
    }else {

        var ele = element.previousSibling;
        while(ele&&ele.nodeType !==1){
            ele = ele.previousSibling;
        }
        return ele;
    }
},
//获得下一个兄弟标签节点
getNextElement:function(element){
    if(element.nextElementSibling){            
        return element.nextElementSibling;
    }else{ 
        var ele = element.nextSibling;
        while(ele && ele.nodeType !==1){                 
        ele= ele.nextSibling;
        }
        return ele; 
    }
},
//获得第一个标签节点
getFirstElement:function(element){
    if(element.firstElementChild){
        return element.firstElementChild;  
    }else{
        var ele =  element.firstChild;  
        while(ele && ele.nodeType !==1){
            ele = ele.nextSibling;
        }
        return ele;
    }
},
//获得最后一个标签节点
getLastElement:function(element){
    if(element.LastElementSibling){  
        return element.LastElementSibling;
        var ele = element.LastSibling;
        while(ele && ele.nodeType !==1){                 
        ele= ele.LastSibling;
        }
        return ele; 
    }
}
}

获得元素的属性值

function getStyle(obj,attr){
    // 能力检测
    if(obj&&obj.currentStyle){
        return  obj.currentStyle[attr];
    }else {
        return  getComputedStyle(obj,null)[attr];
    }
}

scroll 滚轮监控

function scroll(){
    return {   // 直接返回一个对象，只不过对象没有名字而已
    scrollTop: window.pageYOffset || document.documentElement.scrollTop ||         document.body.scrollTop || 0,
    scrollLeft: window.pageXOffset || document.documentElement.scrollLeft ||         document.body.scrollLeft || 0
    }
}

注册事件的方法

function addEventLister(element,type,listener){
    // 能力检测
    if(element.addEventListener){
        element.addEventListener(type,listener,false);
    }else if(element.attachEvent){
        element.attachEvent("on"+type,listener);
    }else {
        element["on"+type] = listener;
    }
}

移除事件

function removeEventListener(obj,type,listener){
    //进行能力检测
    if(obj&&obj.removeEventListener){
        obj.removeEventListener(type,listener,false);
    }else if(obj&&obj.detachEvent){
        obj.detachEvent("on"+type,listener);
    }else {
        obj["on"+type]=null;
    }
}
